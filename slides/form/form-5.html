public function getInputFilter()
{
    if (!$this->inputFilter) {
        $this->inputFilter = new InputFilter();
        $this->inputFilter->add([
            'name' => 'name', // same as name of element
            'required' => 'true',
            'filters' => [['name' => 'StringTrim']],
            'validators' => [[
                'name' => 'StringLength',
                'options' => [
                    'min' => 3,
                    'max' => 64
                ]
            ]],
        ]);
    }
    return $this->inputFilter;
}
